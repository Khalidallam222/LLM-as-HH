```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the heuristic function
    c1, c2 = 0.7, 0.3
    
    # Initialize a dictionary to store the scores for each potential next node
    scores = {}
    
    # Iterate over all unvisited nodes to compute the score
    for node in unvisited_nodes:
        # Look-ahead: estimate the cost if we choose this node as the next one
        look_ahead_cost = np.inf
        for potential_next in unvisited_nodes - {node}:
            cost = distance_matrix[current_node][node] + distance_matrix[node][potential_next]
            if cost < look_ahead_cost:
                look_ahead_cost = cost
        
        # Compute the score for the current node
        # Prioritize minimizing the distance to the next node and the look-ahead cost
        score = c1 * distance_matrix[current_node][node] + c2 * look_ahead_cost
        
        # Store the score for this node
        scores[node] = score
    
    # Select the node with the minimum score as the next node
    next_node = min(scores, key=scores.get)
    
    return next_node
```
This function uses a look-ahead approach by considering the cost of the next possible edge from the chosen node, which helps in making a more informed decision about which node to visit next.
