import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance to the next node and the subsequent node
    scores = {}
    for node in unvisited_nodes:
        # Score based on the direct distance from current to the next node
        direct_distance = distance_matrix[current_node][node]
        # Look-ahead score by considering the distance from the next node to the remaining unvisited nodes
        look_ahead_score = 0
        for next_node in unvisited_nodes - {node}:
            look_ahead_score += distance_matrix[node][next_node]
        # Normalize the look-ahead score by the number of remaining nodes
        look_ahead_score /= len(unvisited_nodes) - 1
        # Score is a combination of the direct distance and the look-ahead score
        scores[node] = direct_distance + look_ahead_score

        # Optionally, penalize the score for the last node if it's not the destination node
        if node != destination_node:
            scores[node] += distance_matrix[node][destination_node]

    # Choose the node with the lowest score
    next_node = min(scores, key=scores.get)
    return next_node
