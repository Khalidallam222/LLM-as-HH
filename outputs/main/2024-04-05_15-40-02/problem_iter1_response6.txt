import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    scores = {}
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the candidate node
        direct_distance = distance_matrix[current_node][node]
        # Look-ahead: estimate the total distance if we visit this node next
        remaining_nodes = unvisited_nodes - {node}
        look_ahead_distance = sum(min(distance_matrix[node][n] for n in remaining_nodes if n != node) for _ in range(len(remaining_nodes) - 1))
        # Combine the direct distance and the look-ahead distance to calculate a score
        score = direct_distance + look_ahead_distance
        scores[node] = score
    # Select the node with the lowest score (shortest total distance)
    next_node = min(scores, key=scores.get)
    return next_node

This `select_next_node_v2` function uses a simple look-ahead mechanism to estimate the total distance if we choose a particular node as the next one. It sums the direct distances from the current node to the candidate node and the minimum distances between the remaining nodes, assuming that the candidate node is chosen next. This should provide a better heuristic for node selection.
