import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the heuristic
    c1, c2 = 0.7, 0.3  # Weights for the distance to the current node and the potential next node's distance to the destination
    
    # Initialize a dictionary to store scores for each potential next node
    scores = {}
    
    # Iterate over the unvisited nodes to calculate scores
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        
        # Calculate the distance from the potential next node to the destination
        to_destination_distance = distance_matrix[node][destination_node]
        
        # Look-ahead: estimate the potential next node's impact on the overall tour length
        remaining_nodes = unvisited_nodes - {node}
        look_ahead_distance = sum(distance_matrix[node][n] for n in remaining_nodes)
        
        # Calculate the score using a weighted sum
        score = c1 * direct_distance + c2 * to_destination_distance + look_ahead_distance
        scores[node] = score
    
    # Select the next node with the lowest score
    next_node = min(scores, key=scores.get)
    
    return next_node
