import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the heuristic
    c1, c2 = 0.7, 0.3  # weights for the current distance and the potential future distance
    
    # Initialize a dictionary to hold scores for each node
    scores = {}
    
    # Iterate over the unvisited nodes to calculate scores
    for node in unvisited_nodes:
        # Calculate the distance from the current node to the potential next node
        current_distance = distance_matrix[current_node][node]
        
        # Look-ahead: estimate the future distance by calculating the mean distance from the potential next node to all remaining nodes
        future_distance = np.mean([distance_matrix[node][n] for n in unvisited_nodes if n != node])
        
        # Calculate the score using the weighted sum of current and future distances
        score = c1 * current_distance - c2 * future_distance
        
        # Save the score in the scores dictionary
        scores[node] = score
    
    # Select the next node with the lowest score
    next_node = min(scores, key=scores.get)
    
    return next_node
