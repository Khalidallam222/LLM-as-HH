```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes with look-ahead mechanism."""
    # Initialize the best score and node
    best_score = float('inf')
    best_node = None
    
    for node in unvisited_nodes:
        # Calculate the direct distance from current to the next node
        direct_distance = distance_matrix[current_node][node]
        
        # Look-ahead: estimate the distance from the next node to the final destination
        remaining_distance = 0
        for remaining_node in unvisited_nodes - {node}:
            remaining_distance += distance_matrix[node][remaining_node]
        remaining_distance += distance_matrix[node][destination_node]
        
        # Calculate the score as a combination of direct distance and look-ahead distance
        score = direct_distance + remaining_distance
        
        # Update the best node if the score is better (lower)
        if score < best_score:
            best_score = score
            best_node = node
            
    return best_node
```
This function uses a simple look-ahead mechanism by summing the direct distance to the next node and an estimate of the remaining distance from that node to the final destination and back to the start. The goal is to minimize the total estimated tour length when choosing the next node.
