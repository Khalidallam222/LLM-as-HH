import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the scoring heuristic
    c1, c2 = 0.6, 0.4  # Weights for the distance to the current node and the potential next node
    
    # Initialize a dictionary to hold the scores for each potential next node
    scores = {}
    
    # Iterate over the unvisited nodes to calculate the score for each
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        
        # Look-ahead: estimate the total distance if we visit this node next
        # by summing the distances from this node to all other unvisited nodes (except itself)
        total_distance_if_chosen = sum(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        
        # Calculate the score as a weighted sum of the direct distance and the total estimated distance
        score = c1 * direct_distance + c2 * total_distance_if_chosen
        
        # Store the score in the dictionary
        scores[node] = score
    
    # Select the node with the lowest score as the next node to visit
    next_node = min(scores, key=scores.get)
    
    return next_node
