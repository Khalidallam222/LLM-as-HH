import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead: prioritize nodes that are closer to the final destination
    # and also consider the distance from the current node to reduce the overall path length.
    look_ahead_factor = 0.7  # weight for the distance to the final destination
    current_distance_factor = 1 - look_ahead_factor  # weight for the distance from the current node
    
    scores = {}
    for node in unvisited_nodes:
        # Calculate the score based on the distance to the current node and the look-ahead distance to the final destination
        score = current_distance_factor * distance_matrix[current_node][node] \
                + look_ahead_factor * distance_matrix[node][destination_node]
        scores[node] = score
    
    # Select the node with the lowest score (best heuristic)
    next_node = min(scores, key=scores.get)
    return next_node
