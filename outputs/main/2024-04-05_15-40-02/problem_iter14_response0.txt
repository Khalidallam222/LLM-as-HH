import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance from the next node to all other unvisited nodes
    scores = {}
    for node in unvisited_nodes:
        # Calculate the sum of distances from the current node to the next node and from the next node to all other unvisited nodes
        look_ahead_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        # Combine this with the direct distance to encourage a balance between short-term and long-term path length
        score = distance_matrix[current_node][node] + 0.5 * look_ahead_distance
        scores[node] = score
    
    # Select the node with the lowest score (which considers both the direct distance and the look-ahead distance)
    next_node = min(scores, key=scores.get)
    return next_node
