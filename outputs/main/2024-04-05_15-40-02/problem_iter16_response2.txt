import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead heuristic: choose the node that minimizes the sum of the distance to it and the next node's distance to the destination
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        lookahead_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if lookahead_distance < min_distance:
            min_distance = lookahead_distance
            next_node = node
    return next_node
