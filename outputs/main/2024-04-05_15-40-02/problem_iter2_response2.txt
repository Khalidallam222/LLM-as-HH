```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance from the potential next node to the remaining unvisited nodes
    def look_ahead_score(node, unvisited):
        return sum(distance_matrix[node][n] for n in unvisited if n != node)

    scores = {}
    for node in unvisited_nodes:
        # Calculate the score based on the direct distance to the current node and the look-ahead score
        direct_distance = distance_matrix[current_node][node]
        look_ahead = look_ahead_score(node, unvisited_nodes)
        scores[node] = direct_distance + look_ahead

    # Select the node with the lowest score (closest in terms of direct distance and look-ahead)
    next_node = min(scores, key=scores.get)
    return next_node
```
This function uses a simple look-ahead strategy where it scores each potential next node based on the sum of distances from that node to all other unvisited nodes. The node with the lowest score (shortest total distance) is selected as the next node to visit.
