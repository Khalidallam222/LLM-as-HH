import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    # Look-ahead mechanism to consider the next best node after the current candidate
    def look_ahead(node):
        # Calculate the score for the given node by looking at the distances to all other unvisited nodes
        scores_from_node = [(distance_matrix[node][next_node], next_node) for next_node in unvisited_nodes if next_node != node]
        # Sort the scores to get the best next node from the current candidate node
        scores_from_node.sort()
        # Take the total distance to the next best node as part of the score
        return distance_matrix[current_node][node] + scores_from_node[0][0] if scores_from_node else float('inf')

    # Calculate the scores for all nodes including the look-ahead score
    scores = {node: look_ahead(node) for node in unvisited_nodes}

    # Select the node with the lowest score (shortest total distance including the look-ahead)
    next_node = min(scores, key=scores.get)
    return next_node

This function `select_next_node_v2` includes a look-ahead mechanism that considers the total distance from the current node to the next node and then to the next best node among the remaining unvisited nodes. This heuristic aims to reduce the overall path length by considering not just the immediate next step but also the potential following step.
