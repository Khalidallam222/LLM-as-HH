import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit from the unvisited nodes using a look-ahead heuristic.
    """
    # Look-ahead by considering the distance from the potential next node to the remaining unvisited nodes
    def look_ahead_score(node):
        return sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)

    # Calculate scores for each unvisited node
    scores = {}
    for node in unvisited_nodes:
        # Combine the direct distance to the current node with the look-ahead score
        score = distance_matrix[current_node][node] + 0.5 * look_ahead_score(node)
        scores[node] = score

    # Select the node with the lowest score
    next_node = min(scores, key=scores.get)
    return next_node
