```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit from the unvisited nodes using a look-ahead heuristic.
    """
    # Look-ahead: choose the next node that minimizes the total distance from the current node
    # to the next node and then to the final destination.
    scores = {}
    for node in unvisited_nodes:
        look_ahead_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        scores[node] = look_ahead_distance

    # In case of a tie, choose the node with the smallest distance to the destination node.
    next_node = min(scores, key=lambda x: (scores[x], distance_matrix[x][destination_node]))
    return next_node
```
