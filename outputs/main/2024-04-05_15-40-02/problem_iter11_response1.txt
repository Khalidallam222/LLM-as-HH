```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead heuristic: choose the node that minimizes the total distance from the current node
    # to the next node and from that next node to the destination node, considering only unvisited nodes.
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        # Evaluate the sum of distances from current to next and from next to destination
        look_ahead_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        # Select the node with the minimum look-ahead distance
        if look_ahead_distance < min_distance:
            min_distance = look_ahead_distance
            next_node = node
    return next_node
```
This function does not use complex scoring mechanisms, but rather a simple look-ahead strategy that tries to minimize the total path distance in the immediate future.
