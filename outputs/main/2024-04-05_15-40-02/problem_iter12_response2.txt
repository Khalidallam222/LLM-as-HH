import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Initialize the score dictionary
    scores = {}
    
    # Iterate over the unvisited nodes to calculate a score for each node
    for node in unvisited_nodes:
        # Calculate the distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        
        # Look-ahead: calculate the distance from the potential next node to the destination node
        look_ahead_distance = distance_matrix[node][destination_node]
        
        # Look-ahead: calculate the minimum distance from the potential next node to the remaining unvisited nodes
        min_distance_to_unvisited = min(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        
        # Calculate the score based on the direct distance, look-ahead distance, and minimum distance to unvisited nodes
        score = direct_distance + 0.5 * look_ahead_distance + 0.5 * min_distance_to_unvisited
        
        # Store the score in the dictionary
        scores[node] = score
    
    # Select the next node with the lowest score
    next_node = min(scores, key=scores.get)
    
    return next_node
