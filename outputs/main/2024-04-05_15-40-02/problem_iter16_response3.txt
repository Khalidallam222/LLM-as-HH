import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance from the potential next node to the remaining unvisited nodes
    scores = {}
    for node in unvisited_nodes:
        # Calculate the total distance if we choose this node as the next one
        total_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        # Look-ahead: sum the distances from this node to all other remaining nodes
        total_distance += sum(distance_matrix[node][n] for n in remaining_nodes)
        # Adjust the score by also considering the distance to the final destination
        total_distance += distance_matrix[node][destination_node]
        scores[node] = total_distance
    
    # Select the node with the minimum estimated total distance
    next_node = min(scores, key=scores.get)
    return next_node
