import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance to the remaining nodes from the potential next node
    scores = {}
    for node in unvisited_nodes:
        remaining_nodes = unvisited_nodes - {node}
        # Calculate the sum of distances from the potential next node to all remaining nodes
        total_distance_from_next = sum(distance_matrix[node][n] for n in remaining_nodes)
        # Calculate the distance from the current node to the potential next node
        current_to_next = distance_matrix[current_node][node]
        # Heuristic: prioritize lower total distance from the next node and shorter distance from the current node
        scores[node] = total_distance_from_next + current_to_next

    # Select the node with the lowest score (which is the sum of distances in this case)
    next_node = min(scores, key=scores.get)
    return next_node
