import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead: choose the next node that minimizes the total distance from the current node to the next node and then to the destination
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        # Evaluate the total distance if we go to 'node' next and then to the 'destination_node'
        total_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if total_distance < min_distance:
            min_distance = total_distance
            next_node = node
    return next_node
