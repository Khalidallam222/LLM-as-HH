```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Assign a score to each unvisited node based on the minimum of the two arcs (current to node, node to destination)
    # and the total distance from the current node to all other unvisited nodes
    scores = {}
    for node in unvisited_nodes:
        # Look-ahead: consider the distance from the current node to the potential next node
        # and the distance from that potential next node to the final destination
        look_ahead_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        # Total distance from the potential next node to all other unvisited nodes
        total_distance_to_unvisited = sum(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        # Score is a weighted combination of look-ahead distance and total distance to unvisited nodes
        scores[node] = look_ahead_distance - 0.5 * total_distance_to_unvisited
    
    # Select the node with the lowest score as the next node to visit
    next_node = min(scores, key=scores.get)
    return next_node
```
This function uses a simple look-ahead heuristic by considering the sum of the distances from the current node to a potential next node and from that node to the final destination, while also penalizing nodes that are far from the remaining unvisited nodes.
