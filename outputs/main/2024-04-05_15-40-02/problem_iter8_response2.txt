import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance from the potential next node to the remaining unvisited nodes
    scores = {}
    for node in unvisited_nodes:
        # Calculate the sum of distances from the current node to the potential next node and from that node to all other unvisited nodes
        total_distance = distance_matrix[current_node][node] + np.sum([distance_matrix[node][n] for n in unvisited_nodes if n != node])
        # Penalize if the next node is the destination node to encourage visiting other nodes first
        if node == destination_node:
            total_distance *= 1.1
        scores[node] = total_distance
    
    # Select the node with the lowest total distance as the next node to visit
    next_node = min(scores, key=scores.get)
    return next_node
