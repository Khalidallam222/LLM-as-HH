import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance from the potential next node to the remaining unvisited nodes
    def look_ahead_score(node, unvisited):
        return sum(distance_matrix[node][n] for n in unvisited if n != node)

    scores = {}
    for node in unvisited_nodes:
        # Calculate the score based on the distance to the current node, the look-ahead score, and the distance to the final destination
        score = distance_matrix[current_node][node] + look_ahead_score(node, unvisited_nodes) - distance_matrix[destination_node][node]
        scores[node] = score

    # Select the node with the lowest score as the next node to visit
    next_node = min(scores, key=scores.get)
    return next_node
