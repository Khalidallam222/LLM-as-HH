import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the heuristic
    c1, c2 = 0.7, 0.3
    
    # Initialize the dictionary to store scores
    scores = {}
    
    # Look-ahead: calculate the score for each node by considering the distance to the node
    # and the minimum distance from that node to the remaining unvisited nodes
    for node in unvisited_nodes:
        # Distance from the current node to the potential next node
        current_to_node = distance_matrix[current_node][node]
        
        # Look-ahead: minimum distance from the potential next node to the remaining unvisited nodes
        min_distance_to_unvisited = min(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        
        # Calculate the score for the node
        score = c1 * current_to_node + c2 * min_distance_to_unvisited
        
        # Store the score in the dictionary
        scores[node] = score
    
    # Select the node with the lowest score as the next node
    next_node = min(scores, key=scores.get)
    
    return next_node
