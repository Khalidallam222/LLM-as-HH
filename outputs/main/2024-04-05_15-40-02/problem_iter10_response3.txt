```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead heuristic: choose the node that minimizes the distance to the remaining nodes
    scores = {}
    for node in unvisited_nodes:
        # Calculate the sum of distances from the current node to the unvisited nodes, excluding the node itself
        remaining_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        # Calculate the distance from the current node to the potential next node
        current_to_node_distance = distance_matrix[current_node][node]
        # Score the node based on the sum of the distance to the next node and the remaining distance
        scores[node] = current_to_node_distance + remaining_distance

    # Select the node with the minimum score (least total distance)
    next_node = min(scores, key=scores.get)
    return next_node
```
This heuristic looks ahead and tries to minimize the total distance by considering both the distance to the next node and the sum of distances from that node to all the remaining unvisited nodes.
