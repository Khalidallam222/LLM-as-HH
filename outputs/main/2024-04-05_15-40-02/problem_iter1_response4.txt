```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    # Define weights for the scoring function
    c1, c2, c3 = 0.5, 0.3, 0.2
    scores = {}
    
    # Iterate over all unvisited nodes
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        
        # Look-ahead: Calculate the minimum distance from the potential next node to all other unvisited nodes
        look_ahead_distance = min(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        
        # Calculate the score for the potential next node
        score = c1 * direct_distance + c2 * look_ahead_distance - c3 * distance_matrix[destination_node][node]
        scores[node] = score

    # Select the next node with the lowest score
    next_node = min(scores, key=scores.get)
    return next_node
```
This version incorporates a look-ahead mechanism by considering the minimum distance from the potential next node to all other unvisited nodes, which helps to potentially reduce the overall path length in the long run.
