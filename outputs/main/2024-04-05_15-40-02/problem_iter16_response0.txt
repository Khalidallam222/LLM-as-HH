```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead heuristic: choose the next node that minimizes the sum of the current distance
    # and the estimated remaining distance to the destination.
    next_node = None
    min_distance = float('inf')
    
    for node in unvisited_nodes:
        # Calculate the current distance to the potential next node
        current_leg_distance = distance_matrix[current_node][node]
        
        # Look-ahead: estimate the remaining distance from the potential next node to the destination
        # by considering the minimum distance to the remaining unvisited nodes plus the distance back to the start
        remaining_distance = min(distance_matrix[node][i] for i in unvisited_nodes if i != node) + distance_matrix[node][destination_node]
        
        # Total estimated distance for choosing this node
        total_estimated_distance = current_leg_distance + remaining_distance
        
        # Choose the node with the minimum estimated total distance
        if total_estimated_distance < min_distance:
            min_distance = total_estimated_distance
            next_node = node
            
    return next_node
```
This heuristic looks ahead to estimate the total remaining distance by considering the minimum distance from the potential next node to all the other unvisited nodes, plus the distance back to the destination node. It then selects the node that minimizes this total estimated distance.
