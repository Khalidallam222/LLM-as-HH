```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the heuristic function
    c1, c2 = 0.7, 0.3
    scores = {}
    
    for node in unvisited_nodes:
        # Look-ahead: calculate the total distance if we choose this node next
        look_ahead_distance = distance_matrix[current_node][node] + np.min([distance_matrix[node][n] for n in unvisited_nodes if n != node])
        # Calculate the score as a weighted sum of the direct distance and the look-ahead distance
        score = c1 * distance_matrix[current_node][node] + c2 * look_ahead_distance
        scores[node] = score
    
    # Choose the node with the lowest score (least potential total distance)
    next_node = min(scores, key=scores.get)
    return next_node
```
This function uses a simple look-ahead mechanism by considering the total potential distance to the next node and from that node to the closest remaining node. This should help guide the selection towards nodes that might lead to a shorter overall path.
