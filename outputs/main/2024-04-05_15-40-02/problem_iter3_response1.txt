import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Define weights for the heuristic function
    c1, c2 = 0.6, 0.4
    
    # Initialize a dictionary to hold the scores for each potential next node
    scores = {}
    
    # Look-ahead: for each potential next node, calculate the cost of the tour if we visit that node next
    for node in unvisited_nodes:
        # Calculate the cost from the current node to the next node
        cost_to_next_node = distance_matrix[current_node][node]
        
        # Calculate the minimum cost from the next node to all other unvisited nodes
        min_cost_to_unvisited = min(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        
        # Calculate the total cost estimate for the path including the next node
        total_cost_estimate = cost_to_next_node + min_cost_to_unvisited
        
        # Calculate the score for the next node, considering both the direct cost and the look-ahead cost
        score = c1 * cost_to_next_node - c2 * total_cost_estimate
        
        # Store the score for each node
        scores[node] = score
    
    # Select the next node with the highest score (最低的代价值)
    next_node = min(scores, key=scores.get)
    
    return next_node
