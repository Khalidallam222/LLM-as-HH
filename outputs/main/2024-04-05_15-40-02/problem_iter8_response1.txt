import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead heuristic: prioritize nodes that bring us closer to the remaining unvisited nodes
    look_ahead_scores = {}
    for node in unvisited_nodes:
        # Calculate the sum of distances from the current node to the potential next node and from that node to all other unvisited nodes
        total_distance = distance_matrix[current_node][node] + sum(distance_matrix[node][next_node] for next_node in unvisited_nodes if next_node != node)
        look_ahead_scores[node] = total_distance

    # Select the node with the smallest look-ahead score
    next_node = min(look_ahead_scores, key=look_ahead_scores.get)

    return next_node
