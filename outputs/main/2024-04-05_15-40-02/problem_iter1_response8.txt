```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes with lookahead mechanism."""
    scores = {}
    for node in unvisited_nodes:
        # Calculate the direct distance from current to the candidate node
        direct_distance = distance_matrix[current_node][node]
        # Look-ahead: estimate the distance from the candidate node to the final destination
        look_ahead_distance = 0
        for n in unvisited_nodes - {node}:
            look_ahead_distance += distance_matrix[node][n]
        # Normalize the look-ahead distance by the number of remaining nodes
        look_ahead_distance /= len(unvisited_nodes) - 1
        # Combine the direct distance and the look-ahead distance to calculate a score
        score = direct_distance + look_ahead_distance
        scores[node] = score
    # Select the node with the minimum score (least total distance)
    next_node = min(scores, key=scores.get)
    return next_node
```
This version introduces a simple lookahead mechanism that estimates the total distance from the current node to the destination by summing the distances from the candidate node to all remaining unvisited nodes, then normalizes this sum by the number of remaining nodes. This lookahead score is combined with the direct distance to give a score for each node, and the node with the lowest score is chosen as the next to visit.
