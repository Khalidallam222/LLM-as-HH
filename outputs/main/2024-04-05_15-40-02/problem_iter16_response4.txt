import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit from the unvisited nodes using a simple look-ahead heuristic.
    This heuristic computes the score for each unvisited node based on the distance to the current node,
    the distance to the destination node, and the number of remaining unvisited nodes connected to it.
    """
    scores = {}
    for node in unvisited_nodes:
        # Score based on the distance to the current node
        score = 1 / (1 + distance_matrix[current_node][node])
        
        # Score adjustment based on the distance to the destination node
        # Encourage nodes that are closer to the destination
        score += 0.5 * (1 / (1 + distance_matrix[node][destination_node]))
        
        # Score adjustment based on the number of remaining connected nodes
        # Encourage nodes that have fewer connections to remaining nodes
        connected_nodes = [n for n in unvisited_nodes if distance_matrix[node][n] > 0]
        score -= len(connected_nodes) * 0.1
        
        scores[node] = score
    
    # Select the node with the highest score
    next_node = max(scores, key=scores.get)
    return next_node
