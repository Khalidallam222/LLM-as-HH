```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead by considering the distance to the remaining nodes from the potential next node
    scores = {}
    for node in unvisited_nodes:
        # Calculate the sum of distances from the current node to the next node and from the next node to all other unvisited nodes
        total_distance = distance_matrix[current_node][node] + np.sum([distance_matrix[node][n] for n in unvisited_nodes if n != node])
        # Penalize if the next node is the destination node, to avoid visiting it too early
        if node == destination_node:
            total_distance += 1000  # Large enough penalty to ensure the destination node is not chosen too early
        scores[node] = total_distance
    
    # Select the node with the minimum total distance as the next node
    next_node = min(scores, key=scores.get)
    return next_node
```
This heuristic looks ahead by summing the distances from the current node to the potential next node and from that next node to all the remaining unvisited nodes. It also penalizes choosing the destination node too early.
