import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead heuristic."""
    # Look-ahead heuristic: consider the distance from the next node to the remaining nodes
    scores = {}
    for node in unvisited_nodes:
        # Compute the sum of distances from this node to all other unvisited nodes
        total_distance_from_node = sum(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        # Add a penalty for not being close to the destination node at the end
        look_ahead_penalty = distance_matrix[node][destination_node]
        # The score is based on the total distance from the node and the distance to the destination
        scores[node] = total_distance_from_node + look_ahead_penalty
    
    # Select the node with the lowest score (least total distance and look-ahead penalty)
    next_node = min(scores, key=scores.get)
    return next_node
