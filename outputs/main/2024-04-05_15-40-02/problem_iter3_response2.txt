```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit from the unvisited nodes using a look-ahead heuristic.
    """
    # Look-ahead heuristic: consider the distance to the next node and the potential next-next node
    scores = {}
    for node in unvisited_nodes:
        # Score based on the distance to the current node and the potential next-next node
        score = distance_matrix[current_node][node] + min(
            distance_matrix[node][next_node] for next_node in unvisited_nodes if next_node != node
        )
        scores[node] = score

    # Select the node with the minimum score (the shortest path)
    next_node = min(scores, key=scores.get)
    return next_node
```
This function uses a simple look-ahead mechanism by adding the minimum distance from the potential next node to any of the remaining unvisited nodes, attempting to minimize the overall path length.
