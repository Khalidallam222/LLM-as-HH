import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes with lookahead mechanism."""
    # Define weights for the scoring heuristic
    c1, c2, c3 = 0.6, 0.3, 0.1
    
    # Initialize scores dictionary
    scores = {}
    
    # Calculate lookahead scores for each unvisited node
    for node in unvisited_nodes:
        # Current distance from the current node to the next node
        current_distance = distance_matrix[current_node][node]
        
        # Lookahead: calculate the distance from the potential next node to the destination
        lookahead_distance = distance_matrix[node][destination_node]
        
        # Lookahead: calculate the minimum distance from the potential next node to the remaining unvisited nodes
        min_distance_to_unvisited = min(distance_matrix[node][i] for i in unvisited_nodes if i != node)
        
        # Calculate the score for the node
        score = c1 * current_distance + c2 * lookahead_distance + c3 * min_distance_to_unvisited
        
        # Assign the score to the node
        scores[node] = score
    
    # Select the node with the minimum score
    next_node = min(scores, key=scores.get)
    
    return next_node
