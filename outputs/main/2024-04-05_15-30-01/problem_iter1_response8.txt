```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    # Initialize a dictionary to store the scores for each node
    scores = {}
    
    # Iterate over all unvisited nodes
    for node in unvisited_nodes:
        # Calculate the cost if we were to visit this node next
        cost_to_node = distance_matrix[current_node][node]
        
        # Look-ahead: estimate the cost from this node to the remaining unvisited nodes
        look_ahead_cost = 0
        for other_node in unvisited_nodes - {node}:
            look_ahead_cost += distance_matrix[node][other_node]
        average_look_ahead_cost = look_ahead_cost / (len(unvisited_nodes) - 1)
        
        # Calculate the score for this node
        # The score is a combination of the cost to visit this node and the average look-ahead cost
        score = cost_to_node + average_look_ahead_cost
        
        # Save the score in the scores dictionary
        scores[node] = score
    
    # Find the node with the minimum score (the least cost)
    next_node = min(scores, key=scores.get)
    
    return next_node
```
This heuristic `select_next_node_v2` considers not only the cost of going to the next node but also the estimated cost of going from that node to the remaining unvisited nodes, which is an attempt to reduce the overall cost of the tour.
