```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    # Initialize the best node and its score
    best_node = None
    best_score = np.inf

    # Iterate over the unvisited nodes
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the candidate node
        direct_distance = distance_matrix[current_node][node]

        # Look-ahead: estimate the distance from the candidate node to the destination node
        # passing through all the other unvisited nodes
        look_ahead_distance = 0
        for other_node in unvisited_nodes - {node}:
            look_ahead_distance += distance_matrix[node][other_node]

        # Calculate the score as a combination of direct distance and look-ahead distance
        score = direct_distance + look_ahead_distance

        # Update the best node if the current node has a better score
        if score < best_score:
            best_score = score
            best_node = node

    return best_node
```
This improved version, `select_next_node_v2`, incorporates a look-ahead mechanism that estimates the total distance from the current node to the destination node through the candidate node, considering all other unvisited nodes. This score is used to determine the best next node to visit.
