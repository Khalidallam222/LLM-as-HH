```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    # Define weights for the scoring
    c1, c2 = 0.6, 0.4
    scores = {}
    
    # Iterate over all unvisited nodes
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        
        # Look-ahead: estimate the distance from the potential next node to the destination node
        # by considering all possible paths through unvisited nodes
        look_ahead_distance = 0
        for next_node in unvisited_nodes - {node}:
            look_ahead_distance += distance_matrix[node][next_node]
        look_ahead_distance /= len(unvisited_nodes) - 1
        
        # Calculate the score for each potential next node
        score = c1 * direct_distance + c2 * look_ahead_distance
        scores[node] = score
    
    # Select the next node with the lowest score
    next_node = min(scores, key=scores.get)
    return next_node
```
This heuristic function `select_next_node_v2` incorporates a look-ahead mechanism by considering the average distance from the potential next node to all other unvisited nodes, which can help in choosing a more globally optimal path.
