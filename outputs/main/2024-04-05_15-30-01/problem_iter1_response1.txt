```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes with lookahead mechanism."""
    
    # Define the weights for the scoring heuristic
    c1, c2, c3 = 0.5, 0.3, 0.2
    
    # Initialize the scores dictionary
    scores = {}
    
    # Iterate over all unvisited nodes
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the next node
        direct_distance = distance_matrix[current_node][node]
        
        # Calculate the lookahead distance from the next node to the destination
        lookahead_distance = 0
        if node != destination_node:
            lookahead_node_scores = []
            for lookahead_node in unvisited_nodes - {node}:
                lookahead_node_scores.append(distance_matrix[node][lookahead_node])
            lookahead_distance = np.min(lookahead_node_scores)
        
        # Calculate the score based on direct distance, lookahead distance, and the distance to the final destination
        score = c1 * direct_distance + c2 * lookahead_distance - c3 * distance_matrix[node][destination_node]
        scores[node] = score
    
    # Select the node with the lowest score as the next node to visit
    next_node = min(scores, key=scores.get)
    
    return next_node
```
This function uses a lookahead mechanism by considering not only the distance from the current node to the next potential node but also the shortest distance from that next node to any other unvisited node (excluding itself). This lookahead helps to potentially reduce the overall path length by considering future steps in the path.
