```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes with a look-ahead mechanism."""
    # Define weights for the scoring function
    c1, c2, c3 = 0.5, 0.3, 0.2
    
    # Initialize the dictionary to store scores for each node
    scores = {}
    
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        
        # Initialize the sum of distances from the potential next node to all other unvisited nodes
        sum_of_distances = 0
        
        # Look-ahead: estimate the total distance if the potential next node is chosen
        for other_node in unvisited_nodes:
            if other_node != node:
                sum_of_distances += distance_matrix[node][other_node]
        
        # Calculate the average distance from the potential next node to all other unvisited nodes
        average_distance_to_unvisited = sum_of_distances / (len(unvisited_nodes) - 1)
        
        # Calculate the score for the potential next node
        score = c1 * direct_distance - c2 * average_distance_to_unvisited + c3 * (distance_matrix[node][destination_node] if node != destination_node else 0)
        
        # Store the score in the scores dictionary
        scores[node] = score
    
    # Select the node with the best score as the next node
    next_node = min(scores, key=scores.get)
    
    return next_node
```
This improved heuristic `select_next_node_v2` includes a look-ahead mechanism that estimates the total distance from the potential next node to all other unvisited nodes, which is then used in the scoring function to determine the best next node to visit.
