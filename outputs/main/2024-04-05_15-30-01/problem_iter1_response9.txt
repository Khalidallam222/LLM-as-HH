```python
import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit from the unvisited nodes using a look-ahead mechanism."""
    scores = {}
    for node in unvisited_nodes:
        # Calculate the direct distance from the current node to the potential next node
        direct_distance = distance_matrix[current_node][node]
        # Initialize the look-ahead score
        look_ahead_score = 0
        # Sum the distances from the potential next node to all other unvisited nodes
        for other_node in unvisited_nodes:
            if other_node != node:
                look_ahead_score += distance_matrix[node][other_node]
        # Average look-ahead distance
        average_look_ahead_distance = look_ahead_score / (len(unvisited_nodes) - 1)
        # The final score is a combination of the direct distance and the average look-ahead distance
        scores[node] = direct_distance + average_look_ahead_distance

        # Optionally, add a penalty for visiting a node that is further away from the destination
        # This encourages the heuristic to prioritize nodes that are closer to the destination
        scores[node] += distance_matrix[node][destination_node] * 0.1

    # Select the node with the smallest score
    next_node = min(scores, key=scores.get)
    return next_node
```
This heuristic looks ahead by considering the sum of distances from the potential next node to all other unvisited nodes, which can help to avoid locally optimal choices that lead to suboptimal paths in the long run.
